# 关于打印测试

  1. 无法打印滚动条下的内容 由于 body 被设置为了 overflow:hidden,打印时高度调整为 auto 在打印时调整为以下内容即可
document.body.style.overflowX = 'auto';
document.body.style.height = 'auto';
window.print();
document.body.style.overflowX = 'hidden';
document.body.style.height = '100%';
  1. table 打印无法自动换行
table {
  page-break-inside: auto !important;
}
tr {
  page-break-inside: avoid !important;
  page-break-after: auto !important;
}
body {
  height: auto;
  overflow: visible;
}
  1. 去除页眉和页脚
@page {
  margin-bottom: 0mm;
  margin-top: 0mm;
}
  1. 打印页面中的部分元素
print() {
      document.body.style.overflow = 'visible';
      const prtHtml = document.getElementById('printEle').innerHTML;
      // Get all stylesheets HTML
      let stylesHtml = '';
      for (const node of [...document.querySelectorAll('link[rel="stylesheet"], style')]) {
        stylesHtml += node.outerHTML;
      }
      // 将样式表及相关内容直接打印在iframe上
      const WinPrint = window.open('', '', 'left=0,top=0,toolbar=0,scrollbars=0,status=0');
      WinPrint.document.write(`<!DOCTYPE html>
      <html>
        <head>
          ${stylesHtml}
        </head>
        <body>
          ${prtHtml}
        </body>
      </html>`);
      WinPrint.document.close();
      WinPrint.focus();
      WinPrint.print();
      WinPrint.close();
      document.body.style.overflowX = 'hidden';
      document.body.style.height = '100%';
    },
